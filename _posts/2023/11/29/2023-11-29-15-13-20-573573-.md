```javascript
// 변수 선언
let name = 'John';
let age = 30;

// 조건문
if (age > 18) {
  console.log(name + '은 성인입니다.');
} else {
  console.log(name + '은 미성년자입니다.');
}

// 반복문
for (let i = 0; i < 5; i++) {
  console.log('반복 횟수: ' + i);
}

// 함수 선언
function greetPerson(name) {
  console.log('안녕하세요, ' + name + '님!');
}

// 함수 호출
greetPerson('Tom');

// 배열
let fruits = ['사과', '바나나', '딸기'];
console.log(fruits[0]); // 사과 출력

// 객체
let person = {
  name: 'John',
  age: 30
};
console.log(person.name); // John 출력
```

위 코드에서는 자바스크립트의 다양한 개념과 문법을 예제로 소개하고 있습니다. 변수 선언, 조건문, 반복문, 함수 선언과 호출, 배열, 객체 등을 다루고 있어 자바스크립트를 공부하는데 도움이 될 수 있을 것입니다.
