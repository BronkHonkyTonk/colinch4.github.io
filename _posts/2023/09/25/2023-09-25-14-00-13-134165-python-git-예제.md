# Git을 사용하여 새로운 저장소 만들기

# 1. Git 초기화 및 저장소 생성
$ git init

# 2. 파일 추가
$ git add 파일명

# 3. 커밋 생성
$ git commit -m "커밋 메시지"

# 4. 원격 저장소 연결
$ git remote add origin 리모트 저장소 주소

# 5. 변경사항 푸시
$ git push -u origin 브랜치명
```
